Step 1 : create view to serialise data


SUCCESS = 'success'
ERROR = 'error'
message_list = ['response', 'status', 'message'] # eg: ["success", 201, "successfully upload the file"]

@csrf_exempt
@api_view(['POST',])
def CreateInspectionReport(request):
	myresponse = {k: [] for k in message_list}
	
	try:
		myData = request.data.copy()
		# print(myData)
		myData['user'] = request.user.id #get the user id
		myData['EmailAddress'] = myData['EmailAddress'] or request.user.email # user enters the email or take from the account

		if not myData['EmailAddress']: # custom validation and response
			RequestResponse(myresponse,ERROR,status.HTTP_400_BAD_REQUEST,{"Email Address" : ["This field cannot be blank"] })
			return Response(data = myresponse)
		try: # check format of date and validate
			myData['InspectionDate'] = datetime.strptime(myData['InspectionDate'], '%m/%d/%Y').date()
		except Exception as e:
			print(e)
			RequestResponse(myresponse,ERROR,status.HTTP_400_BAD_REQUEST,{"Inspection Date" : ["Incorrect Format. It should be mm/dd/yyyy"] })
			return Response(data = myresponse)
		
		try: # check time
			myData['InspectionTime'] = datetime.strptime(request.data['InspectionDate'] + ' ' + myData['InspectionTime'], '%m/%d/%Y %H:%M')
		except Exception as e:
			print(e)
			RequestResponse(myresponse,ERROR,status.HTTP_400_BAD_REQUEST,{"InspectionTime" : ["Incorrect Format. It should be hh:mm !"] })
			return Response(data = myresponse)
		
		try: # check if the property was selected 
			int(myData['InspectionProperty'])
		except Exception as e:
			print(e)
			RequestResponse(myresponse,ERROR,status.HTTP_400_BAD_REQUEST,{"Inspection Property" : ["Select only one property!"] })
			return Response(data = myresponse)
		print(myData)
		
		serializer = serializers.AddInspectionReportSerializer( data=myData) # serialier the data
		if serializer.is_valid():
			try:
				send(**myData.dict()) # create pdf and send email
			except Exception as e:
				print(e)
				RequestResponse(myresponse,ERROR,status.HTTP_400_BAD_REQUEST,{"Email" : ["Could not send mail!"] })
				return Response(data = myresponse)
			account = serializer.save()
			RequestResponse(myresponse,SUCCESS,status.HTTP_201_CREATED, {"Success": [f"Inspection Report e-mailed to {account.EmailAddress}!"] })
			messages.success(request, f'Inspection Report e-mailed to {account.EmailAddress}!')
			# print(serializer)
			return Response(myresponse, )
		RequestResponse(myresponse,ERROR,status.HTTP_400_BAD_REQUEST,serializer.errors)
		return Response(data = myresponse)
	except Exception as e:
		print(e)
		RequestResponse(myresponse,ERROR,status.HTTP_500_INTERNAL_SERVER_ERROR, {"Error": ["Internal Server Error"]})
		return Response(data= myresponse)

step 2: create pdf

def generate_pdf(**Mydata):
	y = 700
	buffer = io.BytesIO() # in memory create pdf
	p = canvas.Canvas(buffer, pagesize=letter)
	p.setFont('Helvetica', 14)
	p.drawString(220,y,   Mydata['InspectionTitle'])
	p.drawString(450, y, "Date:" + timezone.now().strftime('%Y-%b-%d'))
	p.line(30, 675, 550, 675)
	p.drawString(220, y-300, "Time of Inspection: " + str(Mydata['InspectionTime']))
	p.drawString(220, y-200, "Estimated Replacement Cost: " + str(Mydata['EstReplacementCost']))
	p.showPage()
	p.save()
	pdf = buffer.getvalue()
	buffer.close()
	return pdf

step 3: send email using report lab

# settings.py
EMAIL_BACKEND = 'django.core.mail.backends.smtp.EmailBackend'
EMAIL_HOST = "smtp.gmail.com"
EMAIL_HOST_USER = 'your_email@gmail.com'
EMAIL_HOST_PASSWORD = 'your_password'
EMAIL_PORT = 587
EMAIL_USE_TLS = True
# views.py
def send(**Mydata):
	pdf = generate_pdf(**Mydata)
	msg = EmailMessage(Mydata['InspectionTitle'], " Your Inspection Report is ready! ", settings.EMAIL_HOST_USER, to=[Mydata['EmailAddress']])
	msg.attach(f"{Mydata['InspectionTitle']}.pdf", pdf, 'application/pdf')
	msg.content_subtype = "html"
	resp = msg.send()
	print("resp:" , resp)
